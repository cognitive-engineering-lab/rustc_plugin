13pub(crate) fn named(
14    attr: &StructAttr,
15    name: &str,
16    fields: &FieldsNamed,
17    generics: &Generics,
18) -> Result<DerivedTS> {
19    let mut formatted_fields = Vec::new();
20    let mut dependencies = Dependencies::default();
21    if let Some(tag) = &attr.tag {
22        let formatted = format!("{}: \"{}\",", tag, name);
23        formatted_fields.push(quote! {
24            #formatted.to_string()
25        });
26    }
27
28    for field in &fields.named {
29        format_field(
30            &mut formatted_fields,
31            &mut dependencies,
32            field,
33            &attr.rename_all,
34            generics,
35        )?;
36    }
37
38    let fields = quote!(<[String]>::join(&[#(#formatted_fields),*], " "));
39    let generic_args = format_generics(&mut dependencies, generics);
40
41    Ok(DerivedTS {
42        inline: quote! {
43            format!(
44                "{{ {} }}",
45                #fields,
46            )
47        },
48        decl: quote!(format!("interface {}{} {}", #name, #generic_args, Self::inline())),
49        inline_flattened: Some(fields),
50        name: name.to_owned(),
51        dependencies,
52        export: attr.export,
53        export_to: attr.export_to.clone(),
54    })
55}

57// build an expresion which expands to a string, representing a single field of a struct.
58fn format_field(
59    formatted_fields: &mut Vec<TokenStream>,
60    dependencies: &mut Dependencies,
61    field: &Field,
62    rename_all: &Option<Inflection>,
63    generics: &Generics,
64) -> Result<()> {
65    let FieldAttr {
66        type_override,
67        rename,
68        inline,
69        skip,
70        optional,
71        flatten,
72    } = FieldAttr::from_attrs(&field.attrs)?;
73
74    if skip {
75        return Ok(());
76    }
77
78    let (ty, optional_annotation) = match optional {
79        true => (extract_option_argument(&field.ty)?, "?"),
80        false => (&field.ty, ""),
81    };
82
83    if flatten {
84        match (&type_override, &rename, inline) {
85            (Some(_), _, _) => syn_err!("`type` is not compatible with `flatten`"),
86            (_, Some(_), _) => syn_err!("`rename` is not compatible with `flatten`"),
87            (_, _, true) => syn_err!("`inline` is not compatible with `flatten`"),
88            _ => {}
89        }
90
91        formatted_fields.push(quote!(<#ty as ts_rs::TS>::inline_flattened()));
92        dependencies.append_from(ty);
93        return Ok(());
94    }
95
96    let formatted_ty = type_override.map(|t| quote!(#t)).unwrap_or_else(|| {
97        if inline {
98            dependencies.append_from(ty);
99            quote!(<#ty as ts_rs::TS>::inline())
100        } else {
101            format_type(ty, dependencies, generics)
102        }
103    });
104    let field_name = to_ts_ident(field.ident.as_ref().unwrap());
105    let name = match (rename, rename_all) {
106        (Some(rn), _) => rn,
107        (None, Some(rn)) => rn.apply(&field_name),
108        (None, None) => field_name,
109    };
110    let valid_name = raw_name_to_ts_field(name);
111
112    formatted_fields.push(quote! {
113        format!("{}{}: {},", #valid_name, #optional_annotation, #formatted_ty)
114    });
115
116    Ok(())
117}

119fn extract_option_argument(ty: &Type) -> Result<&Type> {
120    match ty {
121        Type::Path(type_path)
122            if type_path.qself.is_none()
123                && type_path.path.leading_colon.is_none()
124                && type_path.path.segments.len() == 1
125                && type_path.path.segments[0].ident == "Option" =>
126        {
127            let segment = &type_path.path.segments[0];
128            match &segment.arguments {
129                PathArguments::AngleBracketed(args) if args.args.len() == 1 => {
130                    match &args.args[0] {
131                        GenericArgument::Type(inner_ty) => Ok(inner_ty),
132                        _ => syn_err!("`Option` argument must be a type"),
133                    }
134                }
135                _ => syn_err!("`Option` type must have a single generic argument"),
136            }
137        }
138        _ => syn_err!("`optional` can only be used on an Option<T> type"),
139    }
140}
