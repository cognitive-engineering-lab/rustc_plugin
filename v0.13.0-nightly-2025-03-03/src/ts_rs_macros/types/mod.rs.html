14pub(crate) fn struct_def(s: &ItemStruct) -> Result<DerivedTS> {

19
20fn type_def(
21    attr: &StructAttr,
22    ident: &Ident,
23    fields: &Fields,
24    generics: &Generics,
25) -> Result<DerivedTS> {
26    let name = attr.rename.clone().unwrap_or_else(|| to_ts_ident(ident));
27    match fields {
28        Fields::Named(named) => match named.named.len() {
29            0 => unit::empty_object(attr, &name),
30            _ => named::named(attr, &name, named, generics),
31        },
32        Fields::Unnamed(unnamed) => match unnamed.unnamed.len() {
33            0 => unit::empty_array(attr, &name),
34            1 => newtype::newtype(attr, &name, unnamed, generics),
35            _ => tuple::tuple(attr, &name, unnamed, generics),
36        },
37        Fields::Unit => unit::null(attr, &name),
38    }
39}
