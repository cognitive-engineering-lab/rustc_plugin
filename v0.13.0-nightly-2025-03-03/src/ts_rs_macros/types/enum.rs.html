13pub(crate) fn r#enum_def(s: &ItemEnum) -> syn::Result<DerivedTS> {
14    let enum_attr: EnumAttr = EnumAttr::from_attrs(&s.attrs)?;
15
16    let name = match &enum_attr.rename {
17        Some(existing) => existing.clone(),
18        None => s.ident.to_string(),
19    };
20
21    if s.variants.is_empty() {
22        return Ok(empty_enum(name, enum_attr));
23    }
24
25    if s.variants.is_empty() {
26        return Ok(DerivedTS {
27            name,
28            inline: quote!("never".to_owned()),
29            decl: quote!("type {} = never;"),
30            inline_flattened: None,
31            dependencies: Dependencies::default(),
32            export: enum_attr.export,
33            export_to: enum_attr.export_to,
34        });
35    }
36
37    let mut formatted_variants = Vec::new();
38    let mut dependencies = Dependencies::default();
39    for variant in &s.variants {
40        format_variant(
41            &mut formatted_variants,
42            &mut dependencies,
43            &enum_attr,
44            variant,
45            &s.generics,
46        )?;
47    }
48
49    let generic_args = format_generics(&mut dependencies, &s.generics);
50    Ok(DerivedTS {
51        inline: quote!([#(#formatted_variants),*].join(" | ")),
52        decl: quote!(format!("type {}{} = {};", #name, #generic_args, Self::inline())),
53        inline_flattened: None,
54        dependencies,
55        name,
56        export: enum_attr.export,
57        export_to: enum_attr.export_to,
58    })
59}

61fn format_variant(
62    formatted_variants: &mut Vec<TokenStream>,
63    dependencies: &mut Dependencies,
64    enum_attr: &EnumAttr,
65    variant: &Variant,
66    generics: &Generics,
67) -> syn::Result<()> {
68    let variant_attr = VariantAttr::from_attrs(&variant.attrs)?;
69
70    if variant_attr.skip {
71        return Ok(());
72    }
73
74    let name = match (variant_attr.rename.clone(), &enum_attr.rename_all) {
75        (Some(rn), _) => rn,
76        (None, None) => variant.ident.to_string(),
77        (None, Some(rn)) => rn.apply(&variant.ident.to_string()),
78    };
79
80    let variant_type = types::type_def(
81        &StructAttr::from(variant_attr),
82        // since we are generating the variant as a struct, it doesn't have a name
83        &format_ident!("_"),
84        &variant.fields,
85        generics,
86    )?;
87    let variant_dependencies = variant_type.dependencies;
88    let inline_type = variant_type.inline;
89
90    let formatted = match enum_attr.tagged()? {
91        Tagged::Untagged => quote!(#inline_type),
92        Tagged::Externally => match &variant.fields {
93            Fields::Unit => quote!(format!("\"{}\"", #name)),
94            Fields::Unnamed(unnamed) if unnamed.unnamed.len() == 1 => {
95                let FieldAttr { skip, .. } = FieldAttr::from_attrs(&unnamed.unnamed[0].attrs)?;
96                if skip {
97                    quote!(format!("\"{}\"", #name))
98                } else {
99                    quote!(format!("{{ \"{}\": {} }}", #name, #inline_type))
100                }
101            }
102            _ => quote!(format!("{{ \"{}\": {} }}", #name, #inline_type)),
103        },
104        Tagged::Adjacently { tag, content } => match &variant.fields {
105            Fields::Unnamed(unnamed) if unnamed.unnamed.len() == 1 => {
106                let FieldAttr {
107                    type_override,
108                    skip,
109                    ..
110                } = FieldAttr::from_attrs(&unnamed.unnamed[0].attrs)?;
111                if skip {
112                    quote!(format!("{{ \"{}\": \"{}\" }}", #tag, #name))
113                } else {
114                    let ty = if let Some(type_override) = type_override {
115                        quote! { #type_override }
116                    } else {
117                        format_type(&unnamed.unnamed[0].ty, dependencies, generics)
118                    };
119                    quote!(format!("{{ \"{}\": \"{}\", \"{}\": {} }}", #tag, #name, #content, #ty))
120                }
121            }
122            Fields::Unit => quote!(format!("{{ \"{}\": \"{}\" }}", #tag, #name)),
123            _ => quote!(
124                format!("{{ \"{}\": \"{}\", \"{}\": {} }}", #tag, #name, #content, #inline_type)
125            ),
126        },
127        Tagged::Internally { tag } => match variant_type.inline_flattened {
128            Some(inline_flattened) => quote! {
129                format!(
130                    "{{ \"{}\": \"{}\", {} }}",
131                    #tag,
132                    #name,
133                    #inline_flattened
134                )
135            },
136            None => match &variant.fields {
137                Fields::Unnamed(unnamed) if unnamed.unnamed.len() == 1 => {
138                    let FieldAttr {
139                        type_override,
140                        skip,
141                        ..
142                    } = FieldAttr::from_attrs(&unnamed.unnamed[0].attrs)?;
143                    if skip {
144                        quote!(format!("{{ \"{}\": \"{}\" }}", #tag, #name))
145                    } else {
146                        let ty = if let Some(type_override) = type_override {
147                            quote! { #type_override }
148                        } else {
149                            format_type(&unnamed.unnamed[0].ty, dependencies, generics)
150                        };
151                        quote!(format!("{{ \"{}\": \"{}\" }} & {}", #tag, #name, #ty))
152                    }
153                }
154                Fields::Unit => quote!(format!("{{ \"{}\": \"{}\" }}", #tag, #name)),
155                _ => {
156                    quote!(format!("{{ \"{}\": \"{}\" }} & {}", #tag, #name, #inline_type))
157                }
158            },
159        },
160    };
161
162    dependencies.append(variant_dependencies);
163    formatted_variants.push(formatted);
164    Ok(())
165}

167// bindings for an empty enum (`never` in TS)
168fn empty_enum(name: impl Into<String>, enum_attr: EnumAttr) -> DerivedTS {
169    let name = name.into();
170    DerivedTS {
171        inline: quote!("never".to_owned()),
172        decl: quote!(format!("type {} = never;", #name)),
173        name,
174        inline_flattened: None,
175        dependencies: Dependencies::default(),
176        export: enum_attr.export,
177        export_to: enum_attr.export_to,
178    }
179}
