1//! Data structures for memoizing computations.
2//!
3//! Contruct new caches using [`Default::default`], then construct/retrieve
4//! elements with [`get`](Cache::get). `get` should only ever be used with one,
5//! `compute` function[^inconsistent].
6//!
7//! In terms of choice,
8//! - [`CopyCache`] should be used for expensive computations that create cheap
9//!   (i.e. small) values.
10//! - [`Cache`] should be used for expensive computations that create expensive
11//!   (i.e. large) values.
12//!
13//! Both types of caches implement **recursion breaking**. In general because
14//! caches are supposed to be used as simple `&` (no `mut`) the reference may be
15//! freely copied, including into the `compute` closure. What this means is that
16//! a `compute` may call [`get`](Cache::get) on the cache again. This is usually
17//! safe and can be used to compute data structures that recursively depend on
18//! one another, dynamic-programming style. However if a `get` on a key `k`
19//! itself calls `get` again on the same `k` this will either create an infinite
20//! recursion or an inconsistent cache[^inconsistent].
21//!
22//! Consider a simple example where we compute the Fibonacci Series with a
23//! [`CopyCache`]:
24//!
25//! ```rs
26//! struct Fib(CopyCache<u32, u32>);
27//!
28//! impl Fib {
29//!   fn get(&self, i: u32) -> u32 {
30//!     self.0.get(i, |_| {
31//!       if this <= 1 {
32//!         return this;
33//!       }
34//!       let fib_1 = self.get(this - 1);
35//!       let fib_2 = self.get(this - 2);
36//!       fib_1 + fib_2
37//!     })
38//!   }
39//! }
40//!
41//! let cache = Fib(Default::default());
42//! let fib_5 = cache.get(5);
43//! ```
44//!
45//! This use of recursive [`get`](CopyCache::get) calls is perfectly legal.
46//! However if we made an error and called `chache.get(this, ...)` (forgetting
47//! the decrement) we would have created an inadvertend infinite recursion.
48//!
49//! To avoid this scenario both caches are implemented to detect when a
50//! recursive call as described is performed and `get` will panic. If your code
51//! uses recursive construction and would like to handle this case gracefully
52//! use [`get_maybe_recursive`](Cache::get_maybe_recursive) instead wich returns
53//! `None` from `get(k)` *iff* `k` this call (potentially transitively)
54//! originates from another `get(k)` call.
55//!
56//! [^inconsistent]: For any given cache value `get` should only ever be used
57//!     with one, referentially transparent `compute` function. Essentially this
58//!     means running `compute(k)` should always return the same value
59//!     *independent of the state of it's environment*. Violation of this rule
60//!     can introduces non-determinism in your program.
61use std::{cell::RefCell, hash::Hash, pin::Pin};
62
63use rustc_data_structures::fx::FxHashMap as HashMap;
64
65/// Cache for non-copyable types.
66pub struct Cache<In, Out>(RefCell<HashMap<In, Option<Pin<Box<Out>>>>>);
67
68impl<In, Out> Cache<In, Out>
69where
70  In: Hash + Eq + Clone,
71{
72  /// Size of the cache
73  pub fn len(&self) -> usize {
74    self.0.borrow().len()
75  }
76  /// Returns the cached value for the given key, or runs `compute` if
77  /// the value is not in cache.
78  ///
79  /// # Panics
80  ///
81  /// If this is a recursive invocation for this key.
82  pub fn get(&self, key: &In, compute: impl FnOnce(In) -> Out) -> &Out {
83    self
84      .get_maybe_recursive(key, compute)
85      .unwrap_or_else(recursion_panic)
86  }
87  /// Returns the cached value for the given key, or runs `compute` if
88  /// the value is not in cache.
89  ///
90  /// Returns `None` if this is a recursive invocation of `get` for key `key`.
91  pub fn get_maybe_recursive<'a>(
92    &'a self,
93    key: &In,
94    compute: impl FnOnce(In) -> Out,
95  ) -> Option<&'a Out> {
96    if !self.0.borrow().contains_key(key) {
97      self.0.borrow_mut().insert(key.clone(), None);
98      let out = Box::pin(compute(key.clone()));
99      self.0.borrow_mut().insert(key.clone(), Some(out));
100    }
101
102    let cache = self.0.borrow();
103    // Important here to first `unwrap` the `Option` created by `get`, then
104    // propagate the potential option stored in the map.
105    let entry = cache.get(key).expect("invariant broken").as_ref()?;
106
107    // SAFETY: because the entry is pinned, it cannot move and this pointer will
108    // only be invalidated if Cache is dropped. The returned reference has a lifetime
109    // equal to Cache, so Cache cannot be dropped before this reference goes out of scope.
110    Some(unsafe { std::mem::transmute::<&'_ Out, &'a Out>(&**entry) })
111  }
112}
113
114fn recursion_panic<A>() -> A {
115  panic!("Recursion detected! The computation of a value tried to retrieve the same from the cache. Using `get_maybe_recursive` to handle this case gracefully.")
116}
117
118impl<In, Out> Default for Cache<In, Out> {
119  fn default() -> Self {
120    Cache(RefCell::new(HashMap::default()))
121  }
122}

124/// Cache for copyable types.
125pub struct CopyCache<In, Out>(RefCell<HashMap<In, Option<Out>>>);
126
127impl<In, Out> CopyCache<In, Out>
128where
129  In: Hash + Eq + Clone,
130  Out: Copy,
131{
132  /// Size of the cache
133  pub fn len(&self) -> usize {
134    self.0.borrow().len()
135  }
136  /// Returns the cached value for the given key, or runs `compute` if
137  /// the value is not in cache.
138  ///
139  /// # Panics
140  ///
141  /// If this is a recursive invocation for this key.
142  pub fn get(&self, key: &In, compute: impl FnOnce(In) -> Out) -> Out {
143    self
144      .get_maybe_recursive(key, compute)
145      .unwrap_or_else(recursion_panic)
146  }
147
148  /// Returns the cached value for the given key, or runs `compute` if
149  /// the value is not in cache.
150  ///
151  /// Returns `None` if this is a recursive invocation of `get` for key `key`.
152  pub fn get_maybe_recursive(
153    &self,
154    key: &In,
155    compute: impl FnOnce(In) -> Out,
156  ) -> Option<Out> {
157    if !self.0.borrow().contains_key(key) {
158      self.0.borrow_mut().insert(key.clone(), None);
159      let out = compute(key.clone());
160      self.0.borrow_mut().insert(key.clone(), Some(out));
161    }
162
163    *self.0.borrow_mut().get(key).expect("invariant broken")
164  }
165}
166
167impl<In, Out> Default for CopyCache<In, Out> {
168  fn default() -> Self {
169    CopyCache(RefCell::new(HashMap::default()))
170  }
171}

173#[cfg(test)]
174mod test {
175  use super::*;
176
177  #[test]
178  fn test_cached() {
179    let cache: Cache<usize, usize> = Cache::default();
180    let x = cache.get(&0, |_| 0);
181    let y = cache.get(&1, |_| 1);
182    let z = cache.get(&0, |_| 2);
183    assert_eq!(*x, 0);
184    assert_eq!(*y, 1);
185    assert_eq!(*z, 0);
186    assert!(std::ptr::eq(x, z));
187  }
188
189  #[test]
190  fn test_recursion_breaking() {
191    struct RecursiveUse(Cache<i32, i32>);
192    impl RecursiveUse {
193      fn get_infinite_recursion(&self, i: i32) -> i32 {
194        self
195          .0
196          .get_maybe_recursive(&i, |_| i + self.get_infinite_recursion(i))
197          .copied()
198          .unwrap_or(-18)
199      }
200      fn get_safe_recursion(&self, i: i32) -> i32 {
201        *self.0.get(&i, |_| {
202          if i == 0 {
203            0
204          } else {
205            self.get_safe_recursion(i - 1) + i
206          }
207        })
208      }
209    }
210
211    let cache = RecursiveUse(Cache::default());
212
213    assert_eq!(cache.get_infinite_recursion(60), 42);
214    assert_eq!(cache.get_safe_recursion(5), 15);
215  }
216}
