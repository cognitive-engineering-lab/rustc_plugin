76/// Programmatically build a rustc compilation session
77pub struct CompileBuilder {
78  input: String,
79  arguments: Vec<String>,
80}
81
82impl CompileBuilder {
83  /// Initialize a compilation from this string of source code.
84  pub fn new(input: impl Into<String>) -> Self {
85    Self {
86      input: input.into(),
87      arguments: vec![],
88    }
89  }
90
91  /// Append additional rustc arguments
92  pub fn with_args(&mut self, args: impl IntoIterator<Item = String>) -> &mut Self {
93    self.arguments.extend(args);
94    self
95  }
96
97  /// Perform the compilation, providing access to it's intermediates state to
98  /// the provided closure
99  pub fn compile(&self, f: impl for<'tcx> FnOnce(CompileResult<'tcx>) + Send) {
100    let mut callbacks = TestCallbacks {
101      input: self.input.clone(),
102      callback: Some(move |tcx: TyCtxt<'_>| f(CompileResult { tcx })),
103    };
104    let args = [
105      "rustc",
106      DUMMY_FILE_NAME,
107      "--crate-type",
108      "lib",
109      "--edition=2024",
110      "-Zidentify-regions",
111      "-Zmir-opt-level=0",
112      "-Zmaximal-hir-to-mir-coverage",
113      "--allow",
114      "warnings",
115      "--sysroot",
116      &*SYSROOT,
117    ]
118    .into_iter()
119    .map(str::to_owned)
120    .chain(self.arguments.clone())
121    .collect::<Box<_>>();
122
123    rustc_driver::catch_fatal_errors(|| {
124      run_compiler(&args, &mut callbacks);
125    })
126    .unwrap();
127  }
128}

130/// Convenience alias for `CompileBuilder::new(input).compile(...)` if the
131/// callback is going to use [`CompileResult::as_body`].
132pub fn compile_body(
133  input: impl Into<String>,
134  callback: impl for<'tcx> FnOnce(TyCtxt<'tcx>, BodyId, &'tcx BodyWithBorrowckFacts<'tcx>)
135    + Send,
136) {
137  CompileBuilder::new(input).compile(|result| {
138    let (body_id, body_with_facts) = result.as_body();
139    callback(result.tcx, body_id, body_with_facts);
140  });
141}

143/// State during the rust compilation. Most of the time you only care about
144/// `self.tcx`, but this wrapper provides additional convenience methods for
145/// getting, e.g. the body of the configured entrypoint.
146pub struct CompileResult<'tcx> {
147  pub tcx: TyCtxt<'tcx>,
148}
149
150impl<'tcx> CompileResult<'tcx> {
151  /// Assume that we compiled only one function and return that function's id and body.
152  pub fn as_body(&self) -> (BodyId, &'tcx BodyWithBorrowckFacts<'tcx>) {
153    let tcx = self.tcx;
154    let (_, body_id) = find_bodies(tcx).remove(0);
155    let def_id = tcx.hir_body_owner_def_id(body_id);
156    let body_with_facts = borrowck_facts::get_body_with_borrowck_facts(tcx, def_id);
157    debug!("{}", body_with_facts.body.to_string(tcx).unwrap());
158    (body_id, body_with_facts)
159  }
160
161  /// Find a body in the target byte range.
162  pub fn as_body_with_range(
163    &self,
164    target: ByteRange,
165  ) -> (BodyId, &'tcx BodyWithBorrowckFacts<'tcx>) {
166    let tcx = self.tcx;
167    let body_id = find_enclosing_bodies(tcx, target.to_span(tcx).unwrap())
168      .next()
169      .unwrap();
170    let def_id = tcx.hir_body_owner_def_id(body_id);
171    let body_with_facts = borrowck_facts::get_body_with_borrowck_facts(tcx, def_id);
172    debug!("{}", body_with_facts.body.to_string(tcx).unwrap());
173
174    (body_id, body_with_facts)
175  }
176}

203pub type RangeMap = HashMap<&'static str, Vec<ByteRange>>;

205pub fn parse_ranges(
206  src: impl AsRef<str>,
207  delimiters: impl AsRef<[(&'static str, &'static str)]>,
208) -> Result<(String, RangeMap)> {
209  let src = src.as_ref();
210  let delimiters = delimiters.as_ref();
211
212  let mut in_idx = 0;
213  let mut out_idx = 0;
214  let mut buf = Vec::new();
215  let bytes = src.bytes().collect::<Vec<_>>();
216  let mut stack = vec![];
217
218  let (opens, closes): (Vec<_>, Vec<_>) = delimiters.iter().copied().unzip();
219  let mut ranges: HashMap<_, Vec<_>> = HashMap::default();
220
221  macro_rules! check_token {
222    ($tokens:expr) => {
223      $tokens
224        .iter()
225        .find(|t| {
226          in_idx + t.len() <= bytes.len()
227            && t.as_bytes() == &bytes[in_idx .. in_idx + t.len()]
228        })
229        .map(|t| *t)
230    };
231  }
232
233  while in_idx < bytes.len() {
234    if let Some(open) = check_token!(opens) {
235      stack.push((out_idx, open));
236      in_idx += open.len();
237      continue;
238    }
239
240    if let Some(close) = check_token!(closes) {
241      let (start, delim) = stack
242        .pop()
243        .with_context(|| anyhow!("Missing open delimiter for \"{close}\""))?;
244      let range = DUMMY_FILE.with(|filename| ByteRange {
245        start: BytePos(start),
246        end: BytePos(out_idx),
247        filename: *filename,
248      });
249      ranges.entry(delim).or_default().push(range);
250      in_idx += close.len();
251      continue;
252    }
253
254    buf.push(bytes[in_idx]);
255    in_idx += 1;
256    out_idx += 1;
257  }
258
259  ensure!(stack.is_empty(), "Unclosed delimiters: {stack:?}");
260
261  let prog_clean = String::from_utf8(buf)?;
262  Ok((prog_clean, ranges))
263}

265pub fn color_ranges(prog: &str, all_ranges: &[(&str, &HashSet<ByteRange>)]) -> String {
266  let mut new_tokens = all_ranges
267    .iter()
268    .flat_map(|(_, ranges)| {
269      ranges.iter().flat_map(|range| {
270        let contained = all_ranges.iter().any(|(_, ranges)| {
271          ranges.iter().any(|other| {
272            range != other && other.start.0 <= range.end.0 && range.end.0 < other.end.0
273          })
274        });
275        let end_marker = if contained { "]" } else { "\x1B[0m]" };
276        [("[\x1B[31m", range.start), (end_marker, range.end)]
277      })
278    })
279    .collect::<Vec<_>>();
280  new_tokens.sort_by_key(|(_, i)| -(isize::try_from(i.0).unwrap()));
281
282  let mut output = prog.to_owned();
283  for (s, i) in new_tokens {
284    output.insert_str(i.0, s);
285  }
286
287  output
288}

290pub fn fmt_ranges(prog: &str, s: &HashSet<ByteRange>) -> String {
291  textwrap::indent(&color_ranges(prog, &[("", s)]), "  ")
292}

294pub fn compare_ranges(
295  expected: &HashSet<ByteRange>,
296  actual: &HashSet<ByteRange>,
297  prog: &str,
298) {
299  let missing = expected - actual;
300  let extra = actual - expected;
301
302  let check = |s: HashSet<ByteRange>, message: &str| {
303    if s.len() > 0 {
304      println!("Expected ranges:\n{}", fmt_ranges(prog, expected));
305      println!("Actual ranges:\n{}", fmt_ranges(prog, actual));
306      panic!("{message} ranges:\n{}", fmt_ranges(prog, &s));
307    }
308  };
309
310  check(missing, "Analysis did NOT have EXPECTED");
311  check(extra, "Actual DID have UNEXPECTED");
312}

339#[derive(Copy, Clone)]
340pub struct PlaceBuilder<'a, 'tcx> {
341  tcx: TyCtxt<'tcx>,
342  body: &'a Body<'tcx>,
343  place: Place<'tcx>,
344}
345
346impl<'tcx> PlaceBuilder<'_, 'tcx> {
347  pub fn field(mut self, i: usize) -> Self {
348    let f = FieldIdx::from_usize(i);
349    let ty = self
350      .place
351      .ty(self.body.local_decls(), self.tcx)
352      .field_ty(self.tcx, f);
353    self.place = self.tcx.mk_place_field(self.place, f, ty);
354    self
355  }
356
357  pub fn deref(mut self) -> Self {
358    self.place = self.tcx.mk_place_deref(self.place);
359    self
360  }
361
362  pub fn downcast(mut self, i: usize) -> Self {
363    let ty = self.place.ty(self.body.local_decls(), self.tcx).ty;
364    let adt_def = ty.ty_adt_def().unwrap();
365    let v = VariantIdx::from_usize(i);
366    self.place = self.tcx.mk_place_downcast(self.place, adt_def, v);
367    self
368  }
369
370  pub fn index(mut self, i: usize) -> Self {
371    self.place = self.tcx.mk_place_index(self.place, Local::from_usize(i));
372    self
373  }
374
375  pub fn mk(self) -> Place<'tcx> {
376    self.place
377  }
378}

380pub fn compare_sets<T: PartialEq + Eq + Clone + Hash + Debug>(
381  expected: impl IntoIterator<Item = T>,
382  actual: impl IntoIterator<Item = T>,
383) {
384  let expected = expected.into_iter().collect::<HashSet<_>>();
385  let actual = actual.into_iter().collect::<HashSet<_>>();
386
387  let missing = &expected - &actual;
388  let extra = &actual - &expected;
389
390  let check = |s: HashSet<T>, message: &str| {
391    if s.len() > 0 {
392      println!(
393        "Expected:\n{}",
394        textwrap::indent(&format!("{expected:#?}"), "  ")
395      );
396      println!(
397        "Actual:\n{}",
398        textwrap::indent(&format!("{actual:#?}"), "  ")
399      );
400      panic!(
401        "{message} ranges:\n{}",
402        textwrap::indent(&format!("{s:#?}"), "  ")
403      );
404    }
405  };
406
407  check(missing, "Result did NOT have EXPECTED");
408  check(extra, "Result DID have UNEXPECTED");
409}
