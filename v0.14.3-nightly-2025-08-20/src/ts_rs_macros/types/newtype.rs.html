1use quote::quote;
2use syn::{FieldsUnnamed, Generics, Result};
3
4use crate::{
5    attr::{FieldAttr, StructAttr},
6    deps::Dependencies,
7    types::generics::{format_generics, format_type},
8    DerivedTS,
9};

11pub(crate) fn newtype(
12    attr: &StructAttr,
13    name: &str,
14    fields: &FieldsUnnamed,
15    generics: &Generics,
16) -> Result<DerivedTS> {
17    if attr.rename_all.is_some() {
18        syn_err!("`rename_all` is not applicable to newtype structs");
19    }
20    if attr.tag.is_some() {
21        syn_err!("`tag` is not applicable to newtype structs");
22    }
23    let inner = fields.unnamed.first().unwrap();
24    let FieldAttr {
25        type_override,
26        rename: rename_inner,
27        inline,
28        skip,
29        optional,
30        flatten,
31    } = FieldAttr::from_attrs(&inner.attrs)?;
32
33    match (&rename_inner, skip, optional, flatten) {
34        (Some(_), ..) => syn_err!("`rename` is not applicable to newtype fields"),
35        (_, true, ..) => return super::unit::null(attr, name),
36        (_, _, true, ..) => syn_err!("`optional` is not applicable to newtype fields"),
37        (_, _, _, true) => syn_err!("`flatten` is not applicable to newtype fields"),
38        _ => {}
39    };
40
41    let inner_ty = &inner.ty;
42    let mut dependencies = Dependencies::default();
43    match (inline, &type_override) {
44        (_, Some(_)) => (),
45        (true, _) => dependencies.append_from(inner_ty),
46        (false, _) => dependencies.push_or_append_from(inner_ty),
47    };
48    let inline_def = match &type_override {
49        Some(o) => quote!(#o.to_owned()),
50        None if inline => quote!(<#inner_ty as ts_rs::TS>::inline()),
51        None => format_type(inner_ty, &mut dependencies, generics),
52    };
53
54    let generic_args = format_generics(&mut dependencies, generics);
55    Ok(DerivedTS {
56        decl: quote!(format!("type {}{} = {};", #name, #generic_args, #inline_def)),
57        inline: inline_def,
58        inline_flattened: None,
59        name: name.to_owned(),
60        dependencies,
61        export: attr.export,
62        export_to: attr.export_to.clone(),
63    })
64}
